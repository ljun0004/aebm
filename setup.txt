# Create environment
conda create -n aebm python=3.12
conda activate aebm

# Install dependencies
python -m pip install --upgrade pip
pip install torch torchvision --index-url https://download.pytorch.org/whl/cu130
pip install tensorboard tqdm scipy einops timm torch-fidelity opencv-python pytorch-lightning omegaconf

# Clone repo
cd [WORK_DIR]
git clone https://github.com/ljun0004/aebm.git

# Download pretrained vae model
cd [VAE_PATH]
wget -c https://ommer-lab.com/files/latent-diffusion/vq-f8-n256.zip

# Download dataset
cd [IMAGENET_PATH]
wget -c --no-check-certificate --user=YOUR_USERNAME --ask-password "https://image-net.org/data/ILSVRC/2012/ILSVRC2012_img_train.tar"
tar -xvf ILSVRC2012_img_train.tar --no-same-owner && rm -f ILSVRC2012_img_train.tar
find . -name "*.tar" | while read -r NAME; do DIR="${NAME%.tar}"; mkdir -p "$DIR"; tar -xvf "$NAME" -C "$DIR" --no-same-owner && rm -f "$NAME"; done

# Cache dataset (use cache_slurm.sh on HPC)
cd [WORK_DIR]
chmod +x cache.sh
./cache.sh

# Training (use train_slurm.sh on HPC)
cd [WORK_DIR]
chmod +x train.sh
./train.sh
